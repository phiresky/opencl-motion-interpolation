cmake_minimum_required(VERSION 3.16.2)

project(MotionInterpolation) 

add_executable(MotionInterpolation)

# ðŸ–•ðŸ–•ðŸ–•ðŸ–•ðŸ–•ðŸ–•

find_package(PNG REQUIRED)

  OpenGL::OpenGL
  OpenGL::GLU
  GLEW::GLEW
  glfw
